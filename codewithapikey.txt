    <script>
        // Store API key in memory only (session-based)
        let userApiKey = '';

        async function analyzeData() {
            const apiKeyInput = document.getElementById('apiKey').value.trim();
            const inputData = document.getElementById('inputData').value.trim();
            const analysisDepth = document.getElementById('analysisDepth').value;
            const contextWeight = document.getElementById('contextWeight').value;
            const analyzeButton = document.getElementById('analyzeButton');
            const loadingIndicator = document.getElementById('loadingIndicator');
            const errorDisplay = document.getElementById('errorDisplay');
            const errorMessage = document.getElementById('errorMessage');
            const resultsContainer = document.getElementById('resultsContainer');

            // Validate API key
            if (!apiKeyInput) {
                errorMessage.textContent = 'Please enter your Anthropic API key';
                errorDisplay.classList.remove('hidden');
                return;
            }

            // Validate input
            if (!inputData) {
                errorMessage.textContent = 'Please enter data to analyze';
                errorDisplay.classList.remove('hidden');
                return;
            }

            // Store API key
            userApiKey = apiKeyInput;

            // Hide error, show loading
            errorDisplay.classList.add('hidden');
            analyzeButton.disabled = true;
            analyzeButton.classList.add('processing');
            analyzeButton.innerHTML = '<span class="flex items-center justify-center"><svg class="animate-spin h-5 w-5 mr-2" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg>Analyzing...</span>';
            loadingIndicator.classList.remove('hidden');

            const systemPrompt = `You are an advanced diagnostic AI assistant specializing in structured data analysis. Your task is to analyze the provided data using a rigorous two-stage methodology.

ANALYSIS FRAMEWORK:

1. DATA SEPARATION & CATEGORIZATION
   - Separate SYMPTOMATIC/OBJECTIVE data (measurements, observations, lab results, error codes, metrics)
   - Separate ENVIRONMENTAL/CONTEXTUAL data (history, lifestyle, system configuration, background factors)

2. TWO-STAGE ANALYSIS
   Stage 1: Generate initial hypotheses based ONLY on symptomatic/objective data
   Stage 2: Refine hypotheses by incorporating contextual/environmental factors

3. CONFIDENCE ASSESSMENT
   - Provide a confidence score (0-100%)
   - Explain the reasoning behind your confidence level
   - Note any data quality issues or gaps

4. RANKED DIFFERENTIAL DIAGNOSES
   - List potential diagnoses/root causes in order of likelihood
   - For each, provide: likelihood rating, supporting evidence, contradicting evidence

5. RECOMMENDATIONS
   - Suggested next steps or actions
   - Priority level for each recommendation

6. MISSING DATA IDENTIFICATION
   - List critical data points needed for higher confidence
   - Explain why each is important

PARAMETERS:
- Analysis Depth: ${analysisDepth}/10 (higher = more rigorous examination)
- Context Weight: ${contextWeight}/10 (higher = more weight on environmental factors)

OUTPUT FORMAT:
Provide your analysis in a clear, structured format using markdown with the following sections:
# Diagnostic Analysis Report
## 1. Data Categorization
## 2. Two-Stage Analysis
## 3. Confidence Assessment
## 4. Differential Diagnoses (Ranked)
## 5. Recommendations
## 6. Critical Missing Data

Be thorough, analytical, and prioritize accuracy over speed.`;

            try {
                const response = await fetch('https://api.anthropic.com/v1/messages', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'x-api-key': userApiKey,
                        'anthropic-version': '2023-06-01'
                    },
                    body: JSON.stringify({
                        model: 'claude-sonnet-4-20250514',
                        max_tokens: 4000,
                        messages: [
                            {
                                role: 'user',
                                content: `${systemPrompt}\n\nDATA TO ANALYZE:\n---\n${inputData}\n---`
                            }
                        ]
                    })
                });

                if (!response.ok) {
                    const errorData = await response.json();
                    throw new Error(errorData.error?.message || `API request failed with status ${response.status}`);
                }

                const data = await response.json();
                const analysisText = data.content
                    .filter(block => block.type === 'text')
                    .map(block => block.text)
                    .join('\n');

                displayResults(analysisText, inputData.length, analysisDepth, contextWeight);
            } catch (error) {
                console.error('Analysis error:', error);
                errorMessage.textContent = error.message || 'An error occurred during analysis';
                errorDisplay.classList.remove('hidden');
            } finally {
                // Reset button
                analyzeButton.disabled = false;
                analyzeButton.classList.remove('processing');
                analyzeButton.textContent = 'Generate Detailed Report';
                loadingIndicator.classList.add('hidden');
            }
        }

        function displayResults(analysisText, inputLength, depth, weight) {
            const resultsContainer = document.getElementById('resultsContainer');
            const timestamp = new Date().toLocaleString();
            
            // Simple markdown to HTML conversion
            let htmlContent = analysisText
                .replace(/^# (.+)$/gm, '<h1 class="text-3xl font-bold text-indigo-700 mt-6 mb-3 border-b-2 border-indigo-200 pb-2">$1</h1>')
                .replace(/^## (.+)$/gm, '<h2 class="text-2xl font-bold text-indigo-600 mt-5 mb-2">$1</h2>')
                .replace(/^### (.+)$/gm, '<h3 class="text-xl font-semibold text-gray-800 mt-4 mb-2">$1</h3>')
                .replace(/^\- (.+)$/gm, '<li class="ml-6 text-gray-700 mb-1">$1</li>')
                .replace(/^\* (.+)$/gm, '<li class="ml-6 text-gray-700 mb-1">$1</li>')
                .replace(/^\d+\. (.+)$/gm, '<li class="ml-6 text-gray-700 mb-1 list-decimal">$1</li>')
                .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
                .replace(/\n\n/g, '</p><p class="text-gray-700 mb-2 leading-relaxed">')
                .replace(/\n/g, '<br>');

            resultsContainer.innerHTML = `
                <div class="bg-white rounded-xl shadow-lg p-8">
                    <div class="flex items-center justify-between mb-6 pb-4 border-b-2 border-indigo-100">
                        <div class="flex items-center">
                            <svg class="w-6 h-6 text-green-500 mr-2" fill="currentColor" viewBox="0 0 20 20">
                                <path fill-rule="evenodd" d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l4-4z" clip-rule="evenodd"></path>
                            </svg>
                            <h2 class="text-2xl font-bold text-indigo-700">Analysis Complete</h2>
                        </div>
                        <span class="text-sm text-gray-500">${timestamp}</span>
                    </div>

                    <div class="prose max-w-none">
                        <p class="text-gray-700 mb-2 leading-relaxed">${htmlContent}</p>
                    </div>

                    <div class="mt-8 pt-6 border-t-2 border-gray-200">
                        <p class="text-sm text-gray-600 mb-2">
                            <strong>Analysis Parameters:</strong> Depth ${depth}/10, Context Weight ${weight}/10
                        </p>
                        <p class="text-sm text-gray-600">
                            <strong>Input Size:</strong> ${inputLength} characters
                        </p>
                        <p class="mt-4 text-sm italic text-red-600 font-medium bg-red-50 p-3 rounded-lg border-l-4 border-red-500">
                            ⚠️ Disclaimer: This report is generated by an AI model for analytical purposes only. It is NOT a substitute for professional judgment, medical diagnosis, or expert consultation.
                        </p>
                    </div>
                </div>
            `;
        }
    </script>
